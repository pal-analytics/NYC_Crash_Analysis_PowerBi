DAX Measures, columns and tables-


1. % Crashes by Factor = 
DIVIDE(COUNTROWS('Vehicle Collisions'),CALCULATE(COUNTROWS('Vehicle Collisions'), ALL('Vehicle Collisions'[Contributing Factor]))

2. Accidents This Year = 
CALCULATE([Total Accidents],FILTER('Date table',YEAR('Date table'[Date])=YEAR(TODAY())))

3. Crash by Factor =
 COUNTROWS('Vehicle Collisions')

4. Injuries per Crash =
 DIVIDE(SUM('Vehicle Collisions'[NUMBER OF CYCLIST INJURED])+SUM('Vehicle Collisions'[NUMBER OF MOTORIST INJURED])+SUM('Vehicle Collisions'[NUMBER OF PEDESTRIANS INJURED])+SUM('Vehicle Collisions'[NUMBER OF PERSONS INJURED]),[Total Accidents])

5. Total Accidents =
 COUNTROWS('Vehicle Collisions')

6. Total Crashes =
 DISTINCTCOUNT('Vehicle Collisions'[COLLISION_ID])

7. Total Injured =
 SUM('Vehicle Collisions'[NUMBER OF PERSONS INJURED])+SUM('Vehicle Collisions'[NUMBER OF CYCLIST INJURED]) +SUM('Vehicle Collisions'[NUMBER OF MOTORIST INJURED])+SUM('Vehicle Collisions'[NUMBER OF PEDESTRIANS INJURED])

8. Total Killed =
 SUM('Vehicle Collisions'[NUMBER OF PERSONS KILLED])+SUM('Vehicle Collisions'[NUMBER OF CYCLIST KILLED]) +SUM('Vehicle Collisions'[NUMBER OF MOTORIST KILLED])+SUM('Vehicle Collisions'[NUMBER OF PEDESTRIANS KILLED])

9. Crash Weekday = 
FORMAT('Date table'[Date],"dddd")

10. Date table =
 CALENDAR(MIN('Vehicle Collisions'[CRASH DATE]),MAX('Vehicle Collisions'[CRASH DATE]))

11. WEEKEND = 
IF(WEEKDAY('Date table'[Date],2)>5,"Weekend", "Weekday")
